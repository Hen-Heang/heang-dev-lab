Sample Form Fields Using Common Codes (공통코드 사용 폼 예시)

// Controller code (컨트롤러 코드)
model.addAttribute("statusCodes", commonCodeService.findByGroup("USER_STATUS"));
model.addAttribute("boardTypeCodes", commonCodeService.findByGroup("BOARD_TYPE"));

// Thymeleaf template (타임리프 템플릿)
<option th:each="code : ${statusCodes}"
        th:value="${code.codeValue}"
